or write ERROR

%------------------------------------------------------------------------------
\nextq
\verb!1 + 2 + 3!
\dotfill
\answerbox{int}

%------------------------------------------------------------------------------
\nextq
\verb!1 * 2 - 3.1!
\dotfill
\answerbox{ERROR}

%------------------------------------------------------------------------------
\nextq
\verb!print_int (3 + 5)!
\dotfill
\answerbox{unit}

%------------------------------------------------------------------------------
\nextq
\verb!3.1415 +. 2.718!
\dotfill
\answerbox{float}

%------------------------------------------------------------------------------
\nextq
\verb!if 1 < 2 then 3 else 4.0!
\dotfill
\answerbox{ERROR}

%------------------------------------------------------------------------------
\nextq
\verb!if 1.1 > 1 then 0.1 else 0.01!
\dotfill
\answerbox{ERROR}

%------------------------------------------------------------------------------
\nextq
\verb!if 2.2 < 3.3 then 1 else 2!
\dotfill
\answerbox{int}

%------------------------------------------------------------------------------
\nextq
\verb!if 42 < 43 then (1 < 2) else (1.1 > 2.2)!
\dotfill
\answerbox{bool}

\nextq

\dotfill
